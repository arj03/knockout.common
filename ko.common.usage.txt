PaginatedViewModel Usage:

== JS ==

function requestExportedLines(curPageIndex, pageSize, curPageIndexObservable) {
    var filterDates = getFilterDates();
    callbackUrl("Default.aspx", "GetExportedCSV", $.toJSON({ pageIndex: curPageIndex, pageSize: pageSize, type: type, 
							     fileFilter: $("#exportedFileFilter").val(), 
                                                             companyFilter: filteredCompany(), from: filterDates[0], to: filterDates[1] }),
                function (d) {
		    if (curPageIndex != curPageIndexObservable())
			return;

                    var data = $.parseJSON(d.d);
                    exportedViewModel.nrElements(data.nrRows);
                    ko.mapping.fromJS(data.rows, exportedViewModel.elements);
                }, function () { showMessage("failed to get exported lines"); });
}

var exportedViewModel = paginatedViewModel(requestExportedLines);
